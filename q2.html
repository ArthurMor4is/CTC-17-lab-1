2.2 Descrição da solução do seguinte problema:¶

O jogo composto de um tabuleiro (7x7). Algumas casas so pretas, algumas das casas pretas tem nmeros. O objetivo iluminar todas clulas brancas, colocando lmpadas nelas. Cada lmpada ilumina sua casa, mas a linha e coluna onde est posicionada, a menos que uma casa preta bloqueie o caminho. Para solucionar o jogo, voc deve satisfazer as seguintes condies:

  • Todas as casas no pretas, esto iluminadas
  • Nenhuma lmpada iluminada por outra lmpada
  • Todas as casas numeradas tem exatamente o nmero indicado de lmpadas adjacentes (as casas acima, abaixo, esquerda e direita)
  • Casas pretas no-numeradas podem ter qualquer nmero de lmpadas adjacentes

2.2.1. Modelagem:¶

  • Foi definida a seguinte conveno entre as possveis configuraes do tabuleiro e nmeros inteiros :

    -2: Espaco vazio

    -1: Parede

    0 a 4: n de lampadas que podem iluminar

    5: Lampada

    6: Blocos iluminados

  • O site indicado no roteiro fornece a opo de salvar o arquivo, aps consultar o cdigo fonte do puzzle para enteder o formato do arquivo, criou-se uma funo pra carregar um puzzle em uma matriz seguindo a conveo acima.
  • Alm disso durante um carregamento gerado um dicionrio para as posies que contm blocos numerados, a chave uma tupla (i, j) e o valor o nmero do bloco, dessa maneira facilitou-se a verificao de posies, pois podemos vericar e atualizar diretamente o dicionrio, esse dicionrio denominado de num_blocks

Foram criadas as seguintes funes:

  • is_valid_position: Verificar todas as condicoes de validade da posio
  • calc_light_cells: Atualiza o valor do grid com o valor correspondente a uma posio iluminada
  • print_map: Desenha o mapa
  • put_position: Coloca uma lmpada na posio caso ela seja vlida
  • hill_climbing: Implementa o algortmo HIll Climbing
  • calc_next_state: Calcula a configurao do prximo estado. Vale destacar que o nmero de iteraes destacados nas solues de cada uma das configuraes iniciais representa quantos estados foram gerados at se chegar nas solues. Um estado se distingue do seu sucessor direto no por uma nica lmpada mas por um conjunto de lmpadas como fica claro no script da funo

def load_puzzle(path):
    file = open(path, "r")
    desc = []
    numered_block = {}
    for line in file:
        text = line.strip().split(":")
        if text[0].strip() == "DESC":
            desc = re.findall('.', text[2].strip())
    grid = np.zeros((7, 7), dtype=int)
    i = 0
    run = 0
    for x in range(7):
        for y in range(7):
            c = '\0'

            if run == 0:
                c = desc[i]
                i = i + 1
                if c >= 'a' and c <= 'z':
                    run = ord(c) - ord('a') + 1

            if run > 0:
                c = 'S'
                run = run - 1

            if c == '0' or c == '1' or c == '2' or c == '3' or c == '4':
                grid[x, y] = ord(c) - ord('0')
                numered_block[(x, y)] = grid[x, y]
            elif c == 'B':
                grid[x, y] = -1
            elif c == 'S':
                grid[x, y] = -2
    return grid, numered_block

def is_valid_position(grid_map, num_block, i, j):
    if (i < 0 or i > 6 or j < 0 or j > 6):
        return False
    
    row = grid_map[:, j]
    col = grid_map[i, :]
    valid = True
    if grid_map[i, j] in range(-1, 6):
        return False
    
    p1 = num_block.get((i+1, j))
    p2 = num_block.get((i-1, j))
    p3 = num_block.get((i, j+1))
    p4 = num_block.get((i, j-1))
    if (p1 is not None) and (p1 == 0):
        return False
    if (p2 is not None) and (p2 == 0):
        return False
    if (p3 is not None) and (p3 == 0):
        return False
    if (p4 is not None) and (p4 == 0):
        return False
    
    for r in range(i+1, 7):
        if row[r] == 5:
            valid = False
        elif row[r] in range(-1, 5):
            break
    for r in range(i-1, -1, -1):
        if row[r] == 5:
            valid = False
        elif row[r] in range(-1, 5):
            break
    for c in range(j+1, 7):
        if col[c] == 5:
            valid = False
        elif col[c] in range(-1, 5):
            break
    for c in range(j-1, -1, -1):
        if col[c] == 5:
            valid = False
        elif col[c] in range(-1, 5):
            break
    return valid
    
def calc_light_cells(grid_sol):
    grids = np.array(grid_sol)
    for i in range(7):
        for j in range(7):
            v = grids[i, j]
            if v == 5:
                for r in range(i, 7):
                    if grids[r, j] == -2:
                        grids[r, j] = 6
                    elif grids[r, j] in range(-1, 5):
                        break
                for r in range(i, -1, -1):
                    if grids[r, j] == -2:
                        grids[r, j] = 6
                    elif grids[r, j] in range(-1, 5):
                        break
                for c in range(j, 7):
                    if grids[i, c] == -2:
                        grids[i, c] = 6
                    elif grids[i, c] in range(-1, 5):
                        break
                for c in range(j, -1, -1):
                    if grids[i, c] == -2:
                        grids[i, c] = 6
                    elif grids[i, c] in range(-1, 5):
                        break
    return grids

def calc_num_spaces(grid_map):
    spaces = (grid_map == -2).sum()
    return spaces

def calc_num_yellow(grid_map):
    lighted = (grid_map == 6).sum()
    lighted = lighted + (grid_map == 5).sum()
    return lighted

def calc_value(grid_map, num_blocks):
    num_spaces = calc_num_spaces(grid_map)
    num_blocks_no_light = sum(list(num_blocks.values()))
    return num_spaces + num_blocks_no_light

def calc_value2(grid_map, num_blocks):
    num_yellows = calc_num_yellow(grid_map)
    num_blocks_no_light = sum(list(num_blocks.values()))
    return num_yellows - 10 * num_blocks_no_light
    
def calc_next_state(grid_map, num_block, num_lights):
    nb = num_block.copy()
    gr = np.array(grid_map)
    nb_item_list = np.array(list(nb.values()))
    idxs = np.argsort(nb_item_list)[::-1]
    ks = list(nb.keys())
    nbb = nb.copy()
    for idx in idxs:
        v = ks[idx]
        pos = []
        pos.append((v[0]+1, v[1]))
        pos.append((v[0]-1, v[1]))
        pos.append((v[0], v[1]+1))
        pos.append((v[0], v[1]-1))
        random.shuffle(pos)
        for p in range(4):
            put_position(gr, nb, pos[p][0], pos[p][1])
    xs = random.choices(list(range(7)), k = num_lights)
    ys = random.choices(list(range(7)), k = num_lights)
    for i in xs:
        for j in ys:
            put_position(gr, nb, i, j)
    
    
    for i in range(7):
        for j in range(7):
            put_position(gr, nb, i, j)
    
                        
    return gr, nb
            
    

def hill_climbing(grid_map, num_blocks, num_iter, num_lights):
    found_max = False
    current, nb = calc_next_state(grid_map, num_blocks, num_lights)
    cur = calc_light_cells(current)
    values = []
    cur_value = calc_value2(cur, nb)
    for i in range(num_iter):
        next_state, nb_next = calc_next_state(grid_map, num_blocks, num_lights)
        ns = calc_light_cells(next_state)
        ns_value = calc_value2(ns, nb_next)
        if ns_value > cur_value:
            current = np.array(next_state)
            nb = nb_next.copy()
            cur = np.array(ns)
            cur_value = ns_value
            if (sum(list(nb.values())) == 0 and (cur == -2).sum() == 0):
                break
            
            
        values.append(cur_value)
        
        if found_max:
            curr = calc_light_cells(cur)
            return curr, nb, values
            
        
    curr = calc_light_cells(cur)
    return curr, nb, values
